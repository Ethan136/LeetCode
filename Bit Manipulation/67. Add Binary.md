class Solution {
public:
    string addBinary(string a, string b) {

        // reverse a & b
        reverse(a.begin(), a.end());
        reverse(b.begin(), b.end());

        // init result
        string result = "";

        // calculate add in binary
        int carry = 0;
        int bitIdx = 0;
        while(carry || (bitIdx < a.size()) || (bitIdx < b.size())) {
            int bit_a = (bitIdx < a.size()) ? (a[bitIdx] - '0') : (0);
            int bit_b = (bitIdx < b.size()) ? (b[bitIdx] - '0') : (0);

            int bit_sum = bit_a + bit_b + carry;
            carry = bit_sum / 2;

            char bit_result = '0' + (bit_sum % 2);
            result = bit_result + result;

            bitIdx++;
        }

        // finish
        return result;
    }
};
