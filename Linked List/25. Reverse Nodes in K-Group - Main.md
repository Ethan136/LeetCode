# LeetCode
[25. Reverse Nodes in K-Group](https://leetcode.com/problems/reverse-nodes-in-k-group/)

# Answer V1 - O(N)：一邊 traverse 一邊 reverse -> 記憶體最佳化，但實作複雜
```Cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
#define DEBUG (0)
#if DEBUG
    void dbgPrintLinkNodes(string title, ListNode* head, int N) {
        cout << title << endl;
        while(N-- && head) {cout << head->val << " "; head = head->next;} cout << endl;
    }
#else
	#define dbgPrintLinkNodes(_DUMMY1_, _DUMMY2_, _DUMMY3_)
#endif

class Solution {
public:

    ListNode* reverseKNodesAndReturnFinishTail(ListNode* beforeStartNode, int k) {
        // structure: beforeStart | [Node1 to NodeK-1] | NodeK (finish Tail) | afterTail
        // return ListNode* = NodeK (finish Tail)

		// dbg info before start
		dbgPrintLinkNodes("Before Reverse", beforeStartNode, k+1);

        // (0) if no need to do reversion
        if(k==1) return beforeStartNode->next;

        // (1) 1st reversed node -> will be linked to the node after the finish tail
        ListNode* revStartNode = beforeStartNode->next;

        // (2) traverse to find the last node to be reversed
        ListNode* lastRevNode = beforeStartNode;
        int travCnt = k;
        while(travCnt--) {
            // if trasverse to the list tail within k-step move after
            if(!lastRevNode || !lastRevNode->next) return NULL;

            // move to next node
            lastRevNode = lastRevNode->next;
        }

        // (3) backup the 1st node after reversed range (after tail node)
        ListNode* AfterTail = lastRevNode->next;

        // (4) 2nd to K-th reverse node -> link back to the previous one
        int backLinkNum = (k-1);
        ListNode* prevNode = revStartNode, *backLinkNode = revStartNode->next;
        while(backLinkNum--) {

            // backup the node next to the backlink node
            ListNode* pNextBackLinkNode = backLinkNode->next;

            // link back to the previous node
            backLinkNode->next = prevNode;

            // update the previous node to the current back link node
            prevNode = backLinkNode;

            // update the backlink node to the one after
            backLinkNode = pNextBackLinkNode;
        }

        // (5) link the node beforestart to the last reversed node
        beforeStartNode->next=lastRevNode;

        // (4) link the reverse start node to the node after reversed range (after tail node)
        revStartNode->next = AfterTail;

		// dbg info after reverse
		dbgPrintLinkNodes("After Reverse", beforeStartNode, k+1);

        // (5) return the reversed tail
        ListNode* revFinishTail = revStartNode;
        return revFinishTail;
    }

    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode augmentHead(0, head);
        ListNode *curNode = &augmentHead;
        while(curNode) {
            curNode = reverseKNodesAndReturnFinishTail(curNode, k);
        }
        return augmentHead.next;
    }
};
```

題目需求
- 把一個 List 每 K 個 Node 一組做反序
- 如果一個 List 處理 N 組反序後 (每組 K Node)，剩餘 Node 不足 K -> 剩餘 Node 不處理

整體架構
- 先設定一個 AugmentHead 連到 input list 前面
- 定義 curNode: 需要 K Node 反序的『前一個 node』 -> 一開始指向 AugmentHead
- 一個 while loop 執行 traverse list to end
	- 裏面每次 traverse 的時候，都從 curNode 開始做 K 個 Node 反序，然後回傳 reversed-K ndoe 的『最後一個』
	- 反序後回傳的 reversed-K node 的最後一個 -> *也就是下次 K Node 反序的前一個* -> 作爲下一次執行的 curNode

K Node 反序列
- input
	- beforeStartNode: K node 反序的前一個 node
		- 作爲左邊界 node
		- *會輸入整體架構中的 curNode*
	- K: 指定幾個 node 反序

- output
	- K node 反序列的最後一個 node -> *如果輸入的 node 往後不足 K個，則回傳 null ptr*

- 執行機制
	- (0) 如果不需要 reverse (k=1) -> 直接回傳下一個 node
	- (1) 找出反序 K node 的第一個: beforeStartNode
	- (2) 找出反序 K node 的最後一個: traverse K times after beforeStartNode
		- 如果 traverse K node 途中 traverse to list end -> *無法反序： return NULL*
	- (3) 找出反序 K node 的下一個: 作爲右邊界 node -> AfterTail
	- (4) 執行 K Node 反序列
		- 概念
			- 處理 K Node 反序 -> **實際上需要 reverse link 的數量是 (k-1)**
			- **從 K Node 的第二個開始，往前 back list**
			- K Node 的第一個: 是 back link 時候最初的 prev Node
			- 每次 Link 後，把當下 traverse node 設爲 prev Node，然後 traverse to next node
		- 流程
			- 每次處理的時候，要先把『下一次 traverse 的 next node 找出來』 -> *這個資訊在 back link 後會遺失*
			- 執行 back link 到 prev node
			- 當前 node 存到 prev node
			- traverse to next node
	- (5) 左邊界 beforeStartNode --連到--> K Node 反序後的 start (就是反序前的 last rev node)
	- (6) K Node 反序後的 tail (也就是反序列前的 start rev node) --連到--> 右邊界 AfterTail
	- (7) 回傳反序列後的 tail node

# Answer V2 - O(2N)：出每一個 sub-reverse-range 的 1st node，拆開來分段 reverse 後再接回去
```Cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {

        // if empty list
        if(!head) return head;

        // if reverse group k = 1 -> no need to do reverse
        if(k <= 1) return head;

        // check the node cnt
        int listNodeCnt = 0;
        for(ListNode *curNode = head; curNode; listNodeCnt++) {
            curNode = curNode->next;
        }

        // check the reversed range cnt
        int revRangeCnt = listNodeCnt / k;
        if(revRangeCnt <= 0) return head;

        // record all the 1st nodes within each reverse range
        vector<ListNode*> revRange1stNodes(revRangeCnt);

        // traverse through the list and record 1st node of all reverse ranges
        ListNode *curNode = head;
        for(int i = 0; i < revRangeCnt; i++) {
            revRange1stNodes[i] = curNode;
            for(int stepCnt = 0; stepCnt < k; stepCnt++) curNode = curNode->next;
        }

        // record the remained un-reversed part
        ListNode *unRevPart1stNode = curNode;

        // do reverse and split all sub-ranges & record 1st & last nodes of the reversed nodes
        vector<ListNode*> revRangeEndNodes(revRange1stNodes.size());
        for(int i = 0; i < revRange1stNodes.size(); i++) {

            // store the reversed tail
            revRangeEndNodes[i] = revRange1stNodes[i];

            // node indicators
            ListNode *prevNode = nullptr;
            ListNode *curNode = revRange1stNodes[i];

            // reverse k nodes within the list
            for(int revCnt = 0; revCnt < k; revCnt++) {
                ListNode *nextNode = curNode->next;
                curNode->next = prevNode;

                prevNode = curNode;
                curNode = nextNode;
            }

            // store the reversed head
            revRange1stNodes[i] = prevNode;
        }

        // merge all reversed sub-lists
        ListNode augmentedHead, *mergedNode = &augmentedHead;
        for(int i = 0; i < revRange1stNodes.size(); i++) {
            mergedNode->next = revRange1stNodes[i];
            mergedNode = revRangeEndNodes[i];
        }

        // link to un-reversed nodes
        mergedNode->next = unRevPart1stNode;

        // finish
        return augmentedHead.next;
    }
};
```
- **直觀會想說，先找「每個 sub-range 的前一個 node 存起來」-> 但這樣是錯的，因為 sub-range 的前一個 node，在前一個 sub-range 做完 reversed 後，就不是這個 sub-range 的前一個 node 了**
	- 比如說 [augment head, 1, 2, 3, 4, 5] k = 2 -> node before sub-range [augment head, 2]
	- sub-range [1, 2] reverse 後 -> [augment head, 2, 1, 3, 4, 5] -> 這時候第二個 sub-range 前一個node 是1，但前面存的卻是「reverse 前的 node 2」

# Answer V3 - O(N)：複習2nd - 模板化寫法（多重模板運用）
```Cpp
class Solution {
private:
    ListNode* getKthNodeFromStart(ListNode *start, int k) {
        ListNode augmentedHead(0, start);

        int travCnt = 0;
        ListNode *travNode = &augmentedHead;
        while(travNode && (travCnt < k)) {
            travNode = travNode->next;
            travCnt++;
        }

        return (travCnt == k) ? (travNode) : (nullptr);
    }
    
    bool cutOffKNodesAfter(ListNode *nodePrevToCutOff, int k, ListNode *&cutOffHead, ListNode *&cutOffTail, ListNode *&nodeAfterCutOff) {
	    if(!nodePrevToCutOff) return false;
	    cutOffHead = nodePrevToCutOff->next;
	    cutOffTail = getKthNodeFromStart(cutOffHead, k);

	    if(!cutOffTail) return false;
	    nodeAfterCutOff = cutOffTail->next;
	    
	    nodePrevToCutOff->next = nullptr;
	    cutOffTail->next = nullptr;
	    return true;
    }
    
    void reverseList(ListNode *head, ListNode *&reverseHead, ListNode *&reverseTail) {
	    ListNode *prev = nullptr;
	    ListNode *node = head;
	    
	    reverseTail = head;
	    while(node) {
		    ListNode *nextNode = node->next;
		    node->next = prev;
		    
		    prev = node;
		    node = nextNode;
	    }
	    reverseHead = prev;
    }

public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        // if there's no need to do reverse
        if(k <= 1) return head;

        // augmented head
        ListNode augmentedHead(0, head);

        // main process
        ListNode *prevNodeToRev = &augmentedHead;
        while(prevNodeToRev) {

	        // cut off k-nodes after prev
            ListNode *cutOffHead, *cutOffTail;
            ListNode* nodeAfterRev;
	        if(!cutOffKNodesAfter(prevNodeToRev, k, cutOffHead, cutOffTail, nodeAfterRev)) break;

            // reverse cut-off nodes
            ListNode *reversedHead, *reversedTail;
            reverseList(cutOffHead, reversedHead, reversedTail);

            // link reversed sub-list back to origin list
            prevNodeToRev->next = reversedHead;
            reversedTail->next = nodeAfterRev;

            // update node traverse for next reverse
            prevNodeToRev = reversedTail;
        }

        // finish
        return augmentedHead.next;
    }
};
```
