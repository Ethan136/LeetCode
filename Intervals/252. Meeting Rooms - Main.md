# 概念
- 先把 interval 排序
- 然後檢查相鄰兩個 interval 有沒有重合

# Answer

## V1: 第一次寫，比較繁瑣
```Cpp
/**
 * Definition of Interval:
 * class Interval {
 * public:
 *     int start, end;
 *     Interval(int start, int end) {
 *         this->start = start;
 *         this->end = end;
 *     }
 * }
 */

class Solution {
public:
    bool canAttendMeetings(vector<Interval>& intervals) {
        // helper function to sort intervals
        auto cmpInterval = [](auto &a, auto &b) {
            return (a.start < b.start) || (a.start == b.start && a.end < b.end);
        };

        // sort intervals by start time
        sort(intervals.begin(), intervals.end(), cmpInterval);

        // helper function to check overlapping
        auto checkIntervalOverlap = [](auto &a, auto &b) {
            if(a.end <= b.start) return false;
            if(b.end <= a.start) return false;
            return true;
        };

        // check interval overlap -> return false if any two intervals are overlapped
        for(int storeIdx = 0, chkIdx = 1; chkIdx < intervals.size(); chkIdx++) {
            if(checkIntervalOverlap(intervals[storeIdx], intervals[chkIdx])) {
                return false;
            }

            storeIdx = chkIdx;
        }

        // for the condition there's no overlap
        return true;
    }
};

```

## V2: 第二次寫，比較簡潔
```Cpp
/**
 * Definition of Interval:
 * class Interval {
 * public:
 *     int start, end;
 *     Interval(int start, int end) {
 *         this->start = start;
 *         this->end = end;
 *     }
 * }
 */

class Solution {
public:
    bool canAttendMeetings(vector<Interval>& intervals) {
        auto cmpIntvl = [](auto &a, auto &b) {
            return a.start < b.start;
        };

        sort(intervals.begin(), intervals.end(), cmpIntvl);

        for(int curIdx = 0, nextIdx = 1; nextIdx < intervals.size(); nextIdx++) {
            if(intervals[curIdx].end > intervals[nextIdx].start) return false;
            curIdx = nextIdx;
        }

        return true;

    }
};
```
