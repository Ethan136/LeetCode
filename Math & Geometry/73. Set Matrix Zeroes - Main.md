# LeetCode
[73. Set Matrix Zeroes](https://leetcode.com/problems/set-matrix-zeroes/)

# Answer：先記錄 1st row / col 是否需要 clear zero，然後檢查剩餘的 Matrix\[r\]\[c\] 並且把需要清零的 row / col 記錄在 1st row / 1st col
## V1：複習1st
```Cpp
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        // row / col num
        int rowNum = matrix.size();
        int colNum = matrix.front().size();

        // check whether col0 / row0 has their own zero
        bool row0AllZero = false;
        for(int c = 0; c < colNum && !row0AllZero; c++) {
            row0AllZero = (matrix[0][c] == 0);            
        }

        bool col0AllZero = false;
        for(int r = 0; r < rowNum && !col0AllZero; r++) {
            col0AllZero = (matrix[r][0] == 0);
        }

        // scan the sub-matrix from matrix[1][1]
        // use col0 / row0 as zero-record-marker
        for(int r = 1; r < rowNum; r++) {
            for(int c = 0; c < colNum; c++) {
                if(matrix[r][c] != 0) continue;
                matrix[r][0] = 0;
                matrix[0][c] = 0;
            }
        }

        // set the sub-region of all zero of sub-matrix
        // from matrix[1][1] to matrix[N-1][N-1]
        for(int r = 1; r < rowNum; r++) {
            for(int c = 1; c < colNum; c++) {
                if(matrix[r][0] && matrix[0][c]) continue;
                matrix[r][c] = 0;
            }
        }

        // deal with row0 & col0
        if(row0AllZero) {
            for(int c = 0; c < colNum; c++) matrix[0][c] = 0;
        }

        if(col0AllZero) {
            for(int r = 0; r < rowNum; r++) matrix[r][0] = 0;
        }
        
    }
};
```


## V2：複習2nd
```Cpp
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        // record whether row0 / col0 need to be cleared
        bool row0Clr = false;
        for(int c = 0; (!row0Clr) && (c < matrix[0].size()); c++) {
            row0Clr = (matrix[0][c] == 0);
        }

        bool col0Clr = false;
        for(int r = 0; (!col0Clr) && (r < matrix.size()); r++) {
            col0Clr = (matrix[r][0] == 0);
        }

        // record clear condition in row0 / col0
        for(int r = 1; r < matrix.size(); r++) {
            for(int c = 1; c < matrix[r].size(); c++) {
                if(matrix[r][c] != 0) continue;
                matrix[r][0] = 0;
                matrix[0][c] = 0;
            }
        }

        // do clear below row0 / col0
        for(int r = 1; r < matrix.size(); r++) {
            for(int c = 1; c < matrix[r].size(); c++) {
                if(matrix[r][0] && matrix[0][c]) continue;
                matrix[r][c] = 0;
            }
        }

        // clear row0 / col0
        for(int c = 0; (row0Clr) && (c < matrix[0].size()); c++) {
            matrix[0][c] = 0;
        }

        for(int r = 0; (col0Clr) && (r < matrix.size()); r++) {
            matrix[r][0] = 0;
        }
    }
};
```
