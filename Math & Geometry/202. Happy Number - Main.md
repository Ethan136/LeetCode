# LeetCode
[202. Happy Number](https://leetcode.com/problems/happy-number/)

# Answer
## 複習1st
```Cpp
class Solution {
public:
    bool isHappy(int n) {
        // lambda func to calc happy num
        auto calcHappy = [](int n) {
            int result = 0;
            while(n) {
                int leastDigit = (n % 10);
                result += leastDigit * leastDigit;

                n = n / 10;
            }
            return result;
        };

        // fast / slow pointer to check loop traverse
        int fast = calcHappy( calcHappy(n) );
        int slow = n;
        while(fast != slow) {
            fast = calcHappy( calcHappy(fast) );
            slow = calcHappy(slow);
        }

        // finish
        return (fast == 1);
    }
};
``` 

## 複習2nd
- **fast-slow ptr 使用上，fast 一定要獨立更新，不能直接拿 slow 再走一次 (2nd)**
```Cpp
class Solution {
public:
    bool isHappy(int n) {
        auto calcHappy = [](int num) {
            int result = 0;
            while(num) {
                int digit = (num%10);
                result += (digit * digit);
                num = num / 10;
            }
            return result;
        };

        int slowCalc = n;
        int fastCalc = calcHappy( calcHappy(n) );
        while( (slowCalc != fastCalc) && (fastCalc != 1) ) {
            slowCalc = calcHappy(slowCalc);
            fastCalc = calcHappy( calcHappy(fastCalc) );
        }

        return (fastCalc == 1);
    }
};
```
