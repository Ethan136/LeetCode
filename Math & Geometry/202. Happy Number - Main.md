# LeetCode
[202. Happy Number](https://leetcode.com/problems/happy-number/)

# Answer
```Cpp
class Solution {
public:
    bool isHappy(int n) {
        // lambda func to calc happy num
        auto calcHappy = [](int n) {
            int result = 0;
            while(n) {
                int leastDigit = (n % 10);
                result += leastDigit * leastDigit;

                n = n / 10;
            }
            return result;
        };

        // fast / slow pointer to check loop traverse
        int fast = calcHappy( calcHappy(n) );
        int slow = n;
        while(fast != slow) {
            fast = calcHappy( calcHappy(fast) );
            slow = calcHappy(slow);
        }

        // finish
        return (fast == 1);
    }
};
```
