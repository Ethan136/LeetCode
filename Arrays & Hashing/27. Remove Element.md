# LeetCode
[27. Remove Element](https://leetcode.com/problems/remove-element/)

# Answer 1st

## 把不同的數字放前面，相同的數字放後面
```Cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        if(nums.size() == 0) return 0;

        int lastIdx = nums.size() - 1;
        int valCnt = 0;
        for(int i = 0; i < (nums.size() - valCnt); ) {
            // not match target value -> check next
            if(nums[i] != val) {
                i++;
                continue;
            }
            // match target value -> swap with backward number
            // -> next term check current idx again
            else {
                swap(nums[i], nums[lastIdx - valCnt]);
                valCnt++;
            }
        }

        return nums.size() - valCnt;
    }
};
```

## 把不同的數字放前面，後續的完全不管他
```Cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int diffCnt = 0;
        for(int i = 0; i < nums.size(); i++) {
            // if match -> not store
            if(nums[i] == val) continue;
        
            // if not match -> store in diff cnt
            nums[diffCnt] = nums[i];
            diffCnt++;
        }

        return diffCnt;
    }
};
```
