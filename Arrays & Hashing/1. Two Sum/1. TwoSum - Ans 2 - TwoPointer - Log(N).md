```Cpp
class Solution {
private:
    struct idxValPair {
        int originIdx;
        int value;
    };
public:
    vector<int> twoSum(vector<int>& nums, int target) {

        vector<idxValPair> idxValPairVect(nums.size());
        for(int i = 0; i < nums.size(); i++) idxValPairVect[i] = {.originIdx = i, .value = nums[i]};

        auto cmpLess = [](idxValPair &a, idxValPair &b) {return a.value < b.value;};
        sort(idxValPairVect.begin(), idxValPairVect.end(), cmpLess);

        for(int l = 0, r = idxValPairVect.size()-1;
                l < r; ) {
            
            int curSum = idxValPairVect[l].value + idxValPairVect[r].value;
            if(curSum == target) return {idxValPairVect[l].originIdx, idxValPairVect[r].originIdx};
            if(curSum > target) r--;
            if(curSum < target) l++;
        }

        return {};
    }
};
```
- sort 後使用 two pointer
- 注意: **要回傳的是 sort 前的 index，所以需要 remap 回 sort 前的 num idx**
