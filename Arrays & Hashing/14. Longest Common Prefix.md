class Solution {
private:
    bool checkStrsWithSameChar(int charIdx, auto &strs) {
        if(charIdx >= strs[0].size()) return false;
        char pivotChar = strs[0][charIdx];

        for(int i = 1; i < strs.size(); i++) {
            if(charIdx >= strs[i].size()) return false;
            if(pivotChar != strs[i][charIdx]) return false;
        }
        return true;
    }
public:
    string longestCommonPrefix(vector<string>& strs) {
        // if there's only one string
        if(strs.size() == 1) return strs[0];

        // init result
        string result = "";

        // compare each idx of each strings
        int charIdx = 0;
        while(true) {
            
            if(!checkStrsWithSameChar(charIdx, strs)) break;

            result += strs[0][charIdx];
            charIdx++;
        }

        // finish
        return result;
    }
};
